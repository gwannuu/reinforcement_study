% 전체 내용을 \raisebox{-4pt}{...}로 감싸서 시각적 중앙을 맞춤
\newcommand{\MethodBlock}[5]{%
    \raisebox{-4pt}{% <--- [여기!] 전체를 -4pt(아래로) 내림. 필요하면 -3pt, -5pt로 조절하세요.
        \begin{tikzpicture}[baseline=-0.4em]
            \node[fill=#1,              
                  minimum size=2.4em,   
                  inner sep=0pt,        
                  text=black,           
                  font=\bfseries\Large, 
                  anchor=center,
                 ] at (0,0) {#2}; 
        \end{tikzpicture}%
        \hspace{#4}%
        \textbf{\large #3}%
        \hspace{#5}%
    }%
}
